[Base Conversion kata from Codewars]
In this kata you ave to implement a base converter, which converts positive integers between arbitrary bases / alphabets. 

								
	convert ('1010', bin, dec)
	
	function convert (input, source, target) {
		let arr = [];
		let inDec = 0;

		for (let i = 0, i < input.length; i ++) {
			len = input.length - 1 - i;
			inDec = source.indexOf(input[len]) *
			      	     Math.pow(source.length, i);			
		}	

		while (inDec >=  target.length) {
			symb = inDec %  target.length;
			arr.unshift(target[symb]);
			inDec = Math.floor(inDec / target.length);
		}

		arr.unshift( target[inDec]);
		return arr.join('')
	}
